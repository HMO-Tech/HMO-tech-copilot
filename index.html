def analyze_image(path: str) -> str:    import base64, anthropic    with open(path, 'rb') as f:        data = base64.b64encode(f.read()).decode()    client = anthropic.Anthropic()    msg = client.messages.create(        model="claude-sonnet-4-20250514",        max_tokens=1024,        messages=[{"role": "user", "content": [            {"type": "image", "source": {                "type": "base64",                "media_type": "image/jpeg",                "data": data}},            {"type": "text", "text": "Describe this image."}        ]}])    return msg.content[0].text

import rhinoscriptsyntax as rs

def move_objects_z(amount=5.0):
    ids = rs.GetObjects("Select objects", preselect=True)
    if not ids: return
    translation = rs.CreateVector(0, 0, amount)
    rs.MoveObjects(ids, translation)

Image Analyzer — Claude API
PY 
def analyze_image(path: str) -> str:
    import base64, anthropic
    with open(path, 'rb') as f:
        data = base64.b64encode(f.read()).decode()
    client = anthropic.Anthropic()
    msg = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=1024,
        messages=[{"role": "user", "content": [
            {"type": "image", "source": {
                "type": "base64",
                "media_type": "image/jpeg",
                "data": data}},
            {"type": "text", "text": "Analyze this design in detail."}
        ]}])
    return msg.content[0].text

Grasshopper — Divide Curve Evenly
GH / PY 
# Grasshopper Python Component
# Inputs: curve (Curve), count (int)
# Outputs: pts (Point3d list), t (float list)
import Rhino.Geometry as rg

def divide_curve(curve, count):
    ok, params = curve.DivideByCount(count, True)
    if not ok:
        return [], []
    points = [curve.PointAt(t) for t in params]
    return points, list(params)
